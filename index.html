var Game_Winner = (S) => 
{
  countA=0;
  countD=0;
  for(i=0;i< S.length;i++){
    if(S[i]==="A"){
      countA++;
    }
    else if(S[i]==="D"){
      countD++;
    }
  }
  if(countA>countD){
    return("Aditya")
  }
  else if(countA< countD){

Palindrome Check

 
var Palin_Check = (str) =>
{
	 N=str.length;
	 count=0;
	 for(i=0;i< (Math.ceil(N/2));i++){
	   if(str.charAt(i)===str.charAt((N-1-i))){
	     count++;
	   }
	 }
	 if(count===(Math.ceil(N/2))){
	   return("True");
	 }
	 else{
	   return("False");
	 }
}

Reverse the String

var Reverse_String = (str) => 
{
  // return(str.split("").reverse().join(""))
  let str2="";
  N=str.length;
  for(i=N-1;i>=0;i--){
    str2+=str[i];
  }
  return str2;
};

var Count_Occ = (s) => 
{
  let n=s.length;
  let news=s.split("").sort().join("");
  let ans="";
  for(i=0;i< n;i++){
    let count=0;
    for(j=0;j< n;j++){
      if(news[i]===news[j] && i>j){
        break;
      }
      if(news[i]===news[j]){
        count++;
      }
    }
    if (count>1){
      ans+=news[i]+count;
    }
  }
 return ans;   
};
